CREATE TABLE MM_DEUDAS(
PERIODO VARCHAR(6),
TIP_REG VARCHAR(10),
COD_SBS_CLI VARCHAR(10),
COD_EMPRESA VARCHAR(4),	
TIP_CREDITO VARCHAR(2),	
COD_CUENTA VARCHAR(14),	
CONDICION VARCHAR(4),	
SALDO VARCHAR(15),	
CLASIFICACION VARCHAR(2),	
SITCRE VARCHAR(2),	
TARJ VARCHAR(2) ,	
SLDVIG VARCHAR(2)
)

SELECT * FROM (
SELECT 1,'PERIODO' AS TIPO,MAX(LENGTH(PERIODO)) MAX ,Min(LENGTH(PERIODO)) MIN  FROM MM_DEUDAS
UNION
SELECT 2,'TIP_REG' AS TIPO,MAX(LENGTH(TIP_REG)) MAX ,Min(LENGTH(TIP_REG)) MIN  FROM MM_DEUDAS
UNION
SELECT 3,'COD_SBS_CLI' AS TIPO,MAX(LENGTH(COD_SBS_CLI)) MAX ,Min(LENGTH(COD_SBS_CLI)) MIN  FROM MM_DEUDAS
UNION
SELECT 4,'COD_EMPRESA' AS TIPO,MAX(LENGTH(COD_EMPRESA)) MAX ,Min(LENGTH(COD_EMPRESA)) MIN  FROM MM_DEUDAS
UNION
SELECT 5,'TIP_CREDITO' AS TIPO,MAX(LENGTH(TIP_CREDITO)) MAX ,Min(LENGTH(TIP_CREDITO)) MIN  FROM MM_DEUDAS
UNION
SELECT 6,'COD_CUENTA' AS TIPO,MAX(LENGTH(COD_CUENTA)) MAX ,Min(LENGTH(COD_CUENTA)) MIN  FROM MM_DEUDAS
UNION
SELECT 7,'CONDICION' AS TIPO,MAX(LENGTH(CONDICION)) MAX ,Min(LENGTH(CONDICION)) MIN  FROM MM_DEUDAS
UNION
SELECT 8,'SALDO' AS TIPO,MAX(LENGTH(SALDO)) MAX ,Min(LENGTH(SALDO)) MIN  FROM MM_DEUDAS
UNION
SELECT 9,'CLASIFICACION' AS TIPO,MAX(LENGTH(CLASIFICACION)) MAX ,Min(LENGTH(CLASIFICACION)) MIN  FROM MM_DEUDAS
UNION
SELECT 10,'SITCRE' AS TIPO,MAX(LENGTH(SITCRE)) MAX ,Min(LENGTH(SITCRE)) MIN  FROM MM_DEUDAS
UNION
SELECT 11,'TARJ' AS TIPO,MAX(LENGTH(TARJ)) MAX ,Min(LENGTH(TARJ)) MIN  FROM MM_DEUDAS
UNION
SELECT 12,'SLDVIG' AS TIPO,MAX(LENGTH(SLDVIG)) MAX ,Min(LENGTH(SLDVIG)) MIN  FROM MM_DEUDAS
)
ORDER BY 1 ASC

SELECT 1,'PERIODO' AS TIPO,COUNT(1)  FROM MM_DEUDAS
WHERE PERIODO IS NULL

SELECT * FROM (

SELECT

MIN(SALDO) AS SALDO_MIN,

MIN(LENGTH(SALDO)) AS SALDO_MINLEN,

COUNT(CASE WHEN CLASIFICACION IS NULL THEN 1 ELSE NULL END) AS CLASIFICACION_ISNULL,
