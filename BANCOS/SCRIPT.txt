CREATE TABLE MM_BANCOS(
codEntidad varchar2(50),
banco varchar2(100),
nombrebanco varchar2(50),
TipoEntidad varchar2(50),
TipoCompetencia varchar2(50),
TipoCompetencia2 varchar2(50)
);

SELECT * FROM MM_BANCOS

SELECT * FROM (
SELECT 1,'codEntidad' AS TIPO,MAX(LENGTH(codEntidad)) MAX ,Min(LENGTH(codEntidad)) MIN  FROM MM_BANCOS
UNION
SELECT 2,'banco' AS TIPO,MAX(LENGTH(banco)) MAX ,Min(LENGTH(banco)) MIN  FROM MM_BANCOS
UNION
SELECT 3,'nombrebanco' AS TIPO,MAX(LENGTH(nombrebanco)) MAX ,Min(LENGTH(nombrebanco)) MIN  FROM MM_BANCOS
UNION
SELECT 4,'TipoEntidad' AS TIPO,MAX(LENGTH(TipoEntidad)) MAX ,Min(LENGTH(TipoEntidad)) MIN  FROM MM_BANCOS
UNION
SELECT 5,'TipoCompetencia' AS TIPO,MAX(LENGTH(TipoCompetencia)) MAX ,Min(LENGTH(TipoCompetencia)) MIN  FROM MM_BANCOS
UNION
SELECT 6,'TipoCompetencia2' AS TIPO,MAX(LENGTH(TipoCompetencia2)) MAX ,Min(LENGTH(TipoCompetencia2)) MIN  FROM MM_BANCOS
)
ORDER BY 1 ASC

SELECT * FROM (
SELECT 1,'codEntidad' AS TIPO,COUNT(1)  FROM MM_BANCOS
WHERE codEntidad IS NULL
UNION
SELECT 2,'banco' AS TIPO,COUNT(1)  FROM MM_BANCOS
WHERE banco IS NULL
UNION
SELECT 3,'nombrebanco' AS TIPO,COUNT(1)  FROM MM_BANCOS
WHERE nombrebanco IS NULL
UNION
SELECT 4,'TipoEntidad' AS TIPO,COUNT(1)  FROM MM_BANCOS
WHERE TipoEntidad IS NULL
UNION
SELECT 5,'TipoCompetencia' AS TIPO,COUNT(1)  FROM MM_BANCOS
WHERE TipoCompetencia IS NULL
UNION
SELECT 6,'TipoCompetencia2' AS TIPO,COUNT(1)  FROM MM_BANCOS
WHERE TipoCompetencia2 IS NULL
)
ORDER BY 1 ASC

SELECT
COUNT(CASE WHEN CODENTIDAD IS NULL THEN 1 ELSE NULL END) AS CODENTIDAD_ISNULL,
MIN(CODENTIDAD) AS CODENTIDAD_MIN,
MAX(CODENTIDAD) AS CODENTIDAD_MAX,
MIN(LENGTH(CODENTIDAD)) AS CODENTIDAD_MINLEN,
MAX(LENGTH(CODENTIDAD)) AS CODENTIDAD_MAXLEN,
COUNT(CASE WHEN BANCO IS NULL THEN 1 ELSE NULL END) AS BANCO_ISNULL,
MIN(BANCO) AS BANCO_MIN,
MAX(BANCO) AS BANCO_MAX,
MIN(LENGTH(BANCO)) AS BANCO_MINLEN,
MAX(LENGTH(BANCO)) AS BANCO_MAXLEN,
COUNT(CASE WHEN NOMBREBANCO IS NULL THEN 1 ELSE NULL END) AS NOMBREBANCO_ISNULL,
MIN(NOMBREBANCO) AS NOMBREBANCO_MIN,
MAX(NOMBREBANCO) AS NOMBREBANCO_MAX,
MIN(LENGTH(NOMBREBANCO)) AS NOMBREBANCO_MINLEN,
MAX(LENGTH(NOMBREBANCO)) AS NOMBREBANCO_MAXLEN,
COUNT(CASE WHEN TIPOENTIDAD IS NULL THEN 1 ELSE NULL END) AS TIPOENTIDAD_ISNULL,
MIN(TIPOENTIDAD) AS TIPOENTIDAD_MIN,
MAX(TIPOENTIDAD) AS TIPOENTIDAD_MAX,
MIN(LENGTH(TIPOENTIDAD)) AS TIPOENTIDAD_MINLEN,
MAX(LENGTH(TIPOENTIDAD)) AS TIPOENTIDAD_MAXLEN,
COUNT(CASE WHEN TIPOCOMPETENCIA IS NULL THEN 1 ELSE NULL END) AS TIPOCOMPETENCIA_ISNULL,
MIN(TIPOCOMPETENCIA) AS TIPOCOMPETENCIA_MIN,
MAX(TIPOCOMPETENCIA) AS TIPOCOMPETENCIA_MAX,
MIN(LENGTH(TIPOCOMPETENCIA)) AS TIPOCOMPETENCIA_MINLEN,
MAX(LENGTH(TIPOCOMPETENCIA)) AS TIPOCOMPETENCIA_MAXLEN,
COUNT(CASE WHEN TIPOCOMPETENCIA2 IS NULL THEN 1 ELSE NULL END) AS TIPOCOMPETENCIA2_ISNULL,
MIN(TIPOCOMPETENCIA2) AS TIPOCOMPETENCIA2_MIN,
MAX(TIPOCOMPETENCIA2) AS TIPOCOMPETENCIA2_MAX,
MIN(LENGTH(TIPOCOMPETENCIA2)) AS TIPOCOMPETENCIA2_MINLEN,
MAX(LENGTH(TIPOCOMPETENCIA2)) AS TIPOCOMPETENCIA2_MAXLEN
FROM MM_BANCOS;
